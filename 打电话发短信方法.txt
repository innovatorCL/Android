case AdUtil.E_CALL_PHONE:  //打电话
                        PackageManager pm = getPackageManager();
                        boolean permission = (PackageManager.PERMISSION_GRANTED == pm.checkPermission("android.permission.CALL_PHONE", "packageName"));
//          boolean permission = PakageUtil.hasPermission(ADActivity.this, android.Manifest.permission.READ_CONTACTS);
                        if(permission){
                            Intent phoneIntent = new Intent(Intent.ACTION_DIAL,Uri.parse(eventValue));
                            phoneIntent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK);
                            startActivity(phoneIntent);
                        }else {
                            FLLogger.i("TAS","没有打电话权限");
                        }
                        break;
                    case AdUtil.E_SEND_EMAIL:	 //发送邮件
                        Intent data=new Intent(Intent.ACTION_SENDTO);
                        String[] datas = eventValue.split("\\?|\\=");
                        if(datas != null && datas.length >0){
                            data.setData(Uri.parse(datas[0]));
                        }
                        if(datas != null && datas.length >2){
                            data.putExtra(Intent.EXTRA_TEXT, datas[2]);
                            startActivity(data);
                        }
                        break;
                    case AdUtil.E_SEND_SMS:	//发送短信
                        String[] sms = eventValue.split("\\:|\\?|\\=");
                        String phone = "";
                        String body = "";
                        if(sms != null && sms.length >1){
                            phone = sms[1];
                        }
                        if(sms != null && sms.length >3){
                            body = sms[3];
                        }
                        Intent smsIntent = new Intent(Intent.ACTION_SENDTO, Uri.parse("smsto:"+phone));
                        smsIntent.putExtra("sms_body", body);
                        startActivity(smsIntent);
                        break;
                    case AdUtil.E_AD_CHANGE:	//广告手动切换
                        break;
                    case AdUtil.E_AD_CLOSE:	//广告关闭
                        break;
                    case AdUtil.E_AD_NO_ACTION:	//不进行互动
                        break;
